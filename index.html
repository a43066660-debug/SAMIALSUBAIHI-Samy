// =============================== // الإعدادات والمتغيرات العامة // =============================== let accountBtc = parseFloat(localStorage.getItem("btc")) || 0.000000017756; let totalAds = parseInt(localStorage.getItem("ads")) || 0; let minerLevel = parseInt(localStorage.getItem("level")) || 1; const BTC_PRICE = 75000; let MINING_INCREMENT = 0.000000000135 + ((minerLevel - 1) * 0.000000000050); let isMiningActive = false; let remainingSeconds = 0; let miningInterval = null; // عناصر DOM const cryptoBalanceEl = document.getElementById("cryptoBalance"); const fiatBalanceEl = document.getElementById("fiatBalance"); const timeCounterEl = document.getElementById("timeCounter"); const hashRateTextEl = document.getElementById("hashRateText"); const startMiningBtn = document.getElementById("startMiningBtn"); const statusBadge = document.getElementById("statusBadge"); const statusText = document.getElementById("statusText"); const adCounterTextEl = document.getElementById("adCounterText"); const levelTextEl = document.getElementById("levelText"); // =============================== // دوال مساعدة // =============================== // حفظ البيانات function saveData() { localStorage.setItem("btc", accountBtc); localStorage.setItem("ads", totalAds); localStorage.setItem("level", minerLevel); } // تحديث الرصيد function refreshMetrics() { cryptoBalanceEl.textContent = accountBtc.toFixed(12); fiatBalanceEl.textContent = "$" + (accountBtc * BTC_PRICE).toFixed(2) + " USD"; } // تنسيق الوقت function formatTime(sec) { const h = String(Math.floor(sec / 3600)).padStart(2, "0"); const m = String(Math.floor((sec % 3600) / 60)).padStart(2, "0"); const s = String(sec % 60).padStart(2, "0"); return `${h}:${m}:${s}`; } // تحديث حالة التعدين function updateMiningUI(active) { if (active) { statusBadge.classList.add("active"); statusText.textContent = "يعمل"; startMiningBtn.disabled = true; } else { statusBadge.classList.remove("active"); statusText.textContent = "متوقف"; startMiningBtn.disabled = false; hashRateTextEl.textContent = "0 TH/s"; } } // =============================== // تشغيل التعدين // =============================== function startMining() { if (isMiningActive) return; isMiningActive = true; remainingSeconds = 6 * 3600; // 6 ساعات updateMiningUI(true); miningInterval = setInterval(() => { if (remainingSeconds <= 0) { stopMining(); return; } remainingSeconds--; timeCounterEl.textContent = formatTime(remainingSeconds); accountBtc += MINING_INCREMENT; refreshMetrics(); saveData(); hashRateTextEl.textContent = (minerLevel * 0.5).toFixed(1) + " TH/s"; }, 1000); } // إيقاف التعدين function stopMining() { clearInterval(miningInterval); isMiningActive = false; updateMiningUI(false); } // =============================== // أزرار التفاعل // =============================== // زر بدء التعدين startMiningBtn.addEventListener("click", startMining); // زر مشاهدة الإعلانات actionAdBtn.addEventListener("click", () => { totalAds++; adCounterTextEl.textContent = totalAds; accountBtc += 0.000000000500; refreshMetrics(); saveData(); }); // زر ترقية العتاد actionUpgradeBtn.addEventListener("click", () => { minerLevel++; levelTextEl.textContent = minerLevel; MINING_INCREMENT = 0.000000000135 + ((minerLevel - 1) * 0.000000000050); saveData(); }); // =============================== // تحميل البيانات عند التشغيل // =============================== refreshMetrics(); adCounterTextEl.textContent = totalAds; levelTextEl.textContent = minerLevel;
